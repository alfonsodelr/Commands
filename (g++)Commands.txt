g++ <filename>.(filetype) 
|| ex. g++ main.cpp : compiles to an a.exe file ||

g++ <filename>.(filetype) -o <filename>.(filetype) 
|| ex. g++ main.cpp -o main.exe : Specifies exe name||

g++ <filename_1>.(filetype_1) <filename_2>.(filetype_2) -o <main_filename>.(filetype)
